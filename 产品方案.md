```
# 伪代码：核心监控逻辑
class JobMonitor:
    def __init__(self):
        self.platforms = {
            'zhubajie': ZhubajieScraper(),
            'codemart': CodemartScraper(),
            'proginn': ProginnScraper(),
        }
        self.db = Database()
        self.notifier = FeishuNotifier()
    
    def run(self):
        for name, scraper in self.platforms.items():
            jobs = scraper.fetch(keywords=user_keywords)
            new_jobs = self.db.filter_new(jobs)
            for job in new_jobs:
                self.notifier.send(job)
            self.db.save(jobs)
```
